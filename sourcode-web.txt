void handleRoot() { server.send(200, "text/html", INDEX_HTML); }

void handleData() {
  String json = "{";
  json += "\"hujan\":\"" + webHujan + "\",";
  json += "\"gelap\":\"" + webGelap + "\",";
  json += "\"atap\":\"" + webAtap + "\"";
  json += "}";
  server.send(200, "application/json", json);
}

// --- TASK 1: MENGURUS SENSOR DAN HARDWARE (Core 1) ---
void TaskHardware(void *pvParameters) {
  for (;;) { // Loop tanpa batas
    int nilaiHujanADC = analogRead(RAIN_SENSOR_PIN); 
    int nilaiCahayaADC = analogRead(LIGHT_SENSOR_PIN);

    bool isRaining = (nilaiHujanADC < THRESHOLD_HUJAN); // Logika ADC Sensor Hujan
    bool isDark = (nilaiCahayaADC > THRESHOLD_GELAP);   // Logika ADC Sensor LDR
    
    webHujan = isRaining ? "Ya (" + String(nilaiHujanADC) + ")" : "Kering (" + String(nilaiHujanADC) + ")";
    webGelap = isDark ? "Gelap (" + String(nilaiCahayaADC) + ")" : "Terang (" + String(nilaiCahayaADC) + ")";

    int targetJemuran = (isRaining || isDark) ? 1 : 0; // 1 = Tutup, 0 = Buka

    if (targetJemuran != statusJemuranSaatIni) {
      if (targetJemuran == 1) {
        Serial.println("EKSEKUSI: Cuaca Buruk. Menutup atap...");
        webAtap = "TUTUP";
        myServo.write(180);
        vTaskDelay(pdMS_TO_TICKS(durasi90Derajat)); // Menggunakan Delay FreeRTOS
        myServo.write(95);
      } else {
        Serial.println("EKSEKUSI: Cuaca Cerah. Membuka atap...");
        webAtap = "BUKA";
        myServo.write(0);
        vTaskDelay(pdMS_TO_TICKS(durasi90Derajat));
        myServo.write(95);
      }
      statusJemuranSaatIni = targetJemuran; 
    }
    
    // Memberi nafas pada Core 1 selama 500ms
    vTaskDelay(pdMS_TO_TICKS(500)); 
  }
}

// --- TASK 2: MENGURUS WEB SERVER (Core 0) ---
void TaskWebServer(void *pvParameters) {
  for (;;) {
    server.handleClient();
    vTaskDelay(pdMS_TO_TICKS(10)); // Jeda sangat kecil agar respons web sangat cepat
  }
}

void setup() {
  Serial.begin(115200); 
  Serial.println("\nSistem Jemuran Pintar ESP32 (RTOS) Dimulai...");

  ESP32PWM::allocateTimer(0);
  myServo.setPeriodHertz(50);
  myServo.attach(SERVO_PIN, 500, 2400); 
  myServo.write(95); 
  delay(1000);

  WiFi.begin(ssid, password);
  Serial.print("Menghubungkan ke WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi Terhubung!");
  Serial.println(WiFi.localIP());

  server.on("/", HTTP_GET, handleRoot);
  server.on("/api/data", HTTP_GET, handleData);
  server.begin();

  // --- INISIALISASI FREERTOS ---
  // Menugaskan TaskWebServer ke Core 0
  xTaskCreatePinnedToCore(
    TaskWebServer, "WebServer_Task", 4096, NULL, 1, NULL, 0 
  );
  
  // Menugaskan TaskHardware ke Core 1
  xTaskCreatePinnedToCore(
    TaskHardware, "Hardware_Task", 2048, NULL, 1, NULL, 1 
  );
}

// loop() dibiarkan kosong karena semua sudah diurus oleh FreeRTOS
void loop() {
  vTaskDelete(NULL); 
}
